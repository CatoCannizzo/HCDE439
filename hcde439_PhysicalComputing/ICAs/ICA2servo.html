const int xIn = A0;
const int yIn = A1;
const int swIn = 0; 
const int servoPin = 3;
int xRead = 0;
int yRead = 0;
int swRead = 0;
int sevOut = 0;
# include <Servo.h> //adds library for servo
Servo myS; //initalizes servo as myS

void setup() {
    // put your setup code here, to run once:
    Serial.begin(9600); //useful for learning
    myS.attach(servoPin);
}

void loop() {
    // put your main code here, to run repeatedly:
    //following chunk reads in the joy stick 
    xRead = analogRead(xIn);
    yRead = analogRead(yIn);
    swRead = digitalRead(swIn);

    //following chunk sends joystick readings to serial
    Serial.print("X: ");
    Serial.print(xRead);
    Serial.print(" | Y: ");
    Serial.print(yRead);
    Serial.print(" | Sw: ");
    Serial.println(swRead);

    //following moves the servo only move when you click the joy stick in
    if (swRead == 0) { //trigger on click
        sevOut = map(yRead, 0, 1023, 0, 180);//maps input and output
        myS.write(sevOut); //outputs to servo
    }
}
